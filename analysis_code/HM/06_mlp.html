cStat <-  function(pred, class, FLIP = TRUE){
  tab <- table(class)
  cind <- wilcox.test(pred ~ class)$statistic/tab[1]/tab[2]
  if(FLIP){
    if(cind < .5) cind <- 1 - cind
  }
  return(cind)
}
